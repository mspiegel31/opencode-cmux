**Structure of the new file:**

```typescript
import type { Plugin } from "@opencode-ai/plugin"
import { EventType } from "./lib/plugin-base"
import { loadConfig } from "./config.js"

export const CmuxPlugin: Plugin = async (ctx) => {
  const { $ } = ctx
  if (!process.env.CMUX_SURFACE_ID) return {}

  const config = await loadConfig()

  // --- cmux help hint (unchanged) ---
  let helpText: string | null = null
  try {
    helpText = await $`cmux help 2>&1`.text()
  } catch { helpText = null }
  const hint = helpText
    ? `You are running inside cmux. The full CLI reference follows.\n\n\`\`\`\n${helpText.trim()}\n\`\`\``
    : "You are running inside cmux. Run `cmux help` to see available commands."
  let hinted = false

  // --- Pending-input tracking ---
  // These Sets hold IDs of in-flight permission requests and questions.
  // isWaitingForInput() guards the idle→done path to prevent spurious done
  // notifications while the session is blocked waiting for user input.
  const pendingPermissions = new Set<string>()
  const pendingQuestions = new Set<string>()

  function isWaitingForInput(): boolean {
    return pendingPermissions.size > 0 || pendingQuestions.size > 0
  }

  // --- Sidebar helpers ---
  // All helpers no-op if config.sidebar.enabled is false.
  // Sidebar key is "opencode" (distinct from subagent "agent-N" keys).

  async function sidebarSetWorking(): Promise<void> {
    if (!config.sidebar.enabled) return
    await $`cmux set-status opencode --icon terminal --color '#f59e0b' --text 'Working'`.quiet()
  }

  async function sidebarSetWaiting(): Promise<void> {
    if (!config.sidebar.enabled) return
    await $`cmux set-status opencode --icon lock --color '#ef4444' --text 'Waiting'`.quiet()
  }

  async function sidebarSetQuestion(): Promise<void> {
    if (!config.sidebar.enabled) return
    await $`cmux set-status opencode --icon help-circle --color '#a855f7' --text 'Question'`.quiet()
  }

  async function sidebarClear(): Promise<void> {
    if (!config.sidebar.enabled) return
    await $`cmux clear-status opencode`.quiet()
  }

  // --- isSubagent helper (unchanged) ---
  async function isSubagent(sessionID: string): Promise<boolean> {
    try {
      const result = await ctx.client.session.get({ path: { id: sessionID } })
      return !!result.data?.parentID
    } catch { return false }
  }

  // --- restoreAfterInputCleared ---
  // After a pending ID is removed, if nothing else is pending:
  //   - If the session is still busy (sessionBusy flag), restore "working"
  //   - If the session is idle, clear sidebar (the done notify already fired or was suppressed)
  // We track the last session status to inform the restore decision.
  let sessionBusy = false

  async function restoreAfterInputCleared(): Promise<void> {
    if (isWaitingForInput()) return  // other pending items remain
    if (sessionBusy) {
      await sidebarSetWorking()
    } else {
      await sidebarClear()
    }
  }

  return {
    event: async ({ event }) => {

      // --- session.status (sidebar working/done transitions) ---
      if (event.type === EventType.SessionStatus) {
        const { sessionID, status } = event.properties
        if (await isSubagent(sessionID)) return
        if (status.type === "busy") {
          sessionBusy = true
          if (!isWaitingForInput()) {
            await sidebarSetWorking()
          }
          // If waiting for input, sidebar already shows waiting/question — leave it.
        } else {
          // idle
          sessionBusy = false
          if (isWaitingForInput()) return  // blocked — don't clear or notify
          await sidebarClear()
          if (config.notify.sessionDone) {
            await $`cmux notify --title "OpenCode" --subtitle "Done" --body "Session finished"`.quiet()
          }
          return
        }
      }

      // --- SessionIdle (legacy fallback — keep for compatibility) ---
      if (event.type === EventType.SessionIdle) {
        const { sessionID } = event.properties
        if (!config.notify.sessionDone) return
        if (await isSubagent(sessionID)) return
        if (isWaitingForInput()) return
        await $`cmux notify --title "OpenCode" --subtitle "Done" --body "Session finished"`.quiet()
      }

      // --- SessionError (unchanged) ---
      if (event.type === EventType.SessionError) {
        const { sessionID } = event.properties
        if (!config.notify.sessionError) return
        if (!sessionID || await isSubagent(sessionID)) return
        await $`cmux notify --title "OpenCode" --subtitle "Error" --body "Session hit an error"`.quiet()
      }

      // --- permission.asked (v2) and permission.updated (v1) ---
      if (
        event.type === EventType.PermissionAsked ||
        event.type === EventType.PermissionUpdated
      ) {
        const { id, sessionID } = event.properties
        if (await isSubagent(sessionID)) return
        if (pendingPermissions.has(id)) return  // already registered by permission.ask hook
        pendingPermissions.add(id)
        await sidebarSetWaiting()
        if (config.notify.permissionRequest) {
          await $`cmux notify --title "OpenCode" --subtitle "Needs Permission" --body "OpenCode is waiting for your approval"`.quiet()
        }
      }

      // --- permission.replied ---
      if (event.type === EventType.PermissionReplied) {
        const { id } = event.properties
        pendingPermissions.delete(id)
        await restoreAfterInputCleared()
      }

      // --- question.asked ---
      if (event.type === EventType.QuestionAsked) {
        const { id, sessionID } = event.properties
        if (await isSubagent(sessionID)) return
        if (pendingQuestions.has(id)) return
        pendingQuestions.add(id)
        const questionText = (event.properties as { questions?: Array<{ header: string }> }).questions?.[0]?.header ?? "OpenCode has a question"
        await sidebarSetQuestion()
        if (config.notify.question) {
          await $`cmux notify --title "OpenCode" --subtitle "Has a Question" --body ${questionText}`.quiet()
        }
      }

      // --- question.replied / question.rejected ---
      if (
        event.type === EventType.QuestionReplied ||
        event.type === EventType.QuestionRejected
      ) {
        const { id } = event.properties
        pendingQuestions.delete(id)
        await restoreAfterInputCleared()
      }

    },

    // Belt-and-suspenders: register permission ID eagerly before the event fires.
    // This means the session.status idle guard sees the pending permission even if
    // the permission.asked event hasn't been processed yet (async ordering edge case).
    "permission.ask": async ({ id, sessionID }: { id: string; sessionID: string }) => {
      if (await isSubagent(sessionID)) return
      pendingPermissions.add(id)
      await sidebarSetWaiting()
      // Desktop notify fires on the event (permission.asked/permission.updated),
      // not here, to avoid double-notification.
    },

    "tui.prompt.append": async (_input: unknown, output: { text?: string }) => {
      if (hinted) return
      hinted = true
      output.text = (output.text || "") + `\n\n[cmux]\n${hint}`
    },
  }
}
```

**Key implementation notes:**
- `sessionBusy` flag tracks last known session.status to inform `restoreAfterInputCleared()`. It starts `false` (no spurious clear on startup).
- The `permission.ask` hook entry must be a string key in the returned hooks object: `"permission.ask": async ({ id, sessionID }) => { ... }`.
- The `questionText` cast is necessary because the OpenCode plugin type system may not expose `questions` in the event properties type — use a defensive cast to `{ questions?: Array<{ header: string }> }` for type safety.
- `pendingPermissions.has(id)` guard in the event handler prevents double-processing when both the hook and the event run (belt-and-suspenders pattern means both will attempt to add).
- All `cmux` CLI calls use `.quiet()` to suppress stdout/stderr noise.
- `try/catch` is NOT needed around individual cmux calls — the existing pattern in this codebase uses `.quiet()` without wrapping (consistent with 04-02 patterns).

Run `npm run typecheck` after writing the file. If there are TypeScript errors around property access on `event.properties`, add appropriate type casts using the `as` keyword — the plugin type system uses a loose union and runtime narrowing is correct even if the TS types are imprecise.
